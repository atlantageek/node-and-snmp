Example SNMP Query

Using the command line tools provided by net-snmp 


% snmpget -v 1 -c public localhost 1.3.6.1.4.1.2021.10.1.3.1
UCD-SNMP-MIB::laLoad.1 = STRING: 0.27

% snmpget -v 1 -c public localhost 1.3.6.1.4.1.2021.10.1.3.1
UCD-SNMP-MIB::laLoad.1 = STRING: 0.23

Using Node.js

Same query in node.js. 


var snmp = require('snmp-native');
var session = new snmp.Session();
var target_oid=[1,3,6,1,4,1,2021,10,1,3,1];
session.get({ oid: target_oid, host: 'localhost', community: 'public' }, 
    function(error, varbinds) {
        if (error) {
            console.log('Fail :(');
        } else {
            console.log(varbinds[0].oid + ' = ' + varbinds[0].value + ' (' + varbinds[0].type + ')');
        }
});

function query_snmp() {
    var session = new snmp.Session();
    session.get({ oid: [1, 3, 6, 1, 4, 1, 2021,10,1,3,1], host: 'localhost', community: 'public' }, function(error, varbinds) {
        if (error) {
            console.log('Fail :(');

        }
    });
}
